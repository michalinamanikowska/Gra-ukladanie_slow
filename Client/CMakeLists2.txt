cmake_minimum_required(VERSION 3.5)

project(QtClient LANGUAGES CXX)

set(CMAKE_INCLUDE_CURRENT_DIR ON)
find_package(Qt5 5.7 COMPONENTS Core Gui Widgets Network REQUIRED)
add_executable(Client
    main.cpp
    window.cpp
    client.cpp
    window.ui
    window.h
    client.h
)
target_link_libraries(Client PRIVATE Qt5::Core Qt5::Gui Qt5::Widgets Qt5::Network)
target_compile_definitions(Client PRIVATE QT_NO_CAST_FROM_ASCII)
set_target_properties(Client PROPERTIES
	AUTOMOC ON
	AUTOUIC ON
	AUTORCC ON
	CXX_STANDARD 11
	CXX_STANDARD_REQUIRED ON
	VERSION "1.0.0"
	EXPORT_NAME Client
	ARCHIVE_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/lib"
	LIBRARY_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/lib"
	RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin"
)
